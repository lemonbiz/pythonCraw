CREATE TABLE `mt_directory` (
  `ID` int(11) NOT NULL AUTO_INCREMENT,
  `FIRST_LEVEL_DIRECTORY` varchar(255) COLLATE utf8_bin DEFAULT NULL COMMENT '一级目录',
  `SECOND_LEVEL_DIRECTORY` varchar(255) COLLATE utf8_bin DEFAULT NULL COMMENT '二级目录',
  `URL` varchar(255) COLLATE utf8_bin DEFAULT NULL,
  `DIRECTORY_CODE` varchar(255) COLLATE utf8_bin DEFAULT NULL COMMENT '目录代码',
  `INBASE_TIME` datetime DEFAULT NULL COMMENT '入库时间',
  `UPDATE_TIME` datetime DEFAULT NULL COMMENT '更新时间',
  `IS_DELETE` tinyint(3) NOT NULL DEFAULT '0' COMMENT '删除状态：1删除，0未删除',
  PRIMARY KEY (`ID`),
  KEY `DIRECTORY_CODE` (`DIRECTORY_CODE`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8 COLLATE=utf8_bin
